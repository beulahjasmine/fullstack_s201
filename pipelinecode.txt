pipeline {
    agent any

    tools {
        jdk 'JDK_HOME'
        maven 'MAVEN_HOME'
    }

    environment {
        TOMCAT_URL  = 'http://13.49.209.132:9090/manager/text'
        MAIN_REPO   = 'https://github.com/beulahjasmine/fullstack_s201.git'
        BACKEND_DIR = 'crud_back'
        FRONTEND_DIR = 'crud_front'
        BACKEND_WAR = 'crud_back/target/springapp1.war'
        FRONTEND_WAR = 'crud_front/target/frontapp1.war'
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: "${MAIN_REPO}"
            }
        }

        stage('Build Backend') {
            steps {
                dir("${BACKEND_DIR}") {
                    sh "mvn clean package"
                }
            }
        }

        stage('Build Frontend') {
            steps {
                dir("${FRONTEND_DIR}") {
                    sh "mvn clean package"
                }
            }
        }

        stage('Deploy to Tomcat') {
            steps {
                withCredentials([usernamePassword(credentialsId: 'tomcat-creds', usernameVariable: 'TOMCAT_USER', passwordVariable: 'TOMCAT_PASS')]) {
                    sh """
                        curl -u $TOMCAT_USER:$TOMCAT_PASS --upload-file ${BACKEND_WAR} "${TOMCAT_URL}/deploy?path=/backend&update=true"
                        curl -u $TOMCAT_USER:$TOMCAT_PASS --upload-file ${FRONTEND_WAR} "${TOMCAT_URL}/deploy?path=/frontend&update=true"
                    """
                }
            }
        }
    }
}
